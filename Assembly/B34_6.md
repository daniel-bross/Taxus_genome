# 1. create HiC-phased contig assembly with hifiasm

```bash
hic_dir=../../data/raw-data/Illumina-HiC
pb_dir=../../data/raw-data/PacBio
out_dir=../../data/outputs/$(basename $(pwd))

## set purge level to 1
## purge levels 1-3 were tested;  level 1 performed best based on contiguity, number of contigs and busco scores
l=2
mkdir -p ${out_dir}

hifiasm -l ${l} -o vpTaxBacc_B34-6.hifiasm.asm --seed 123 -t 48 -f39 --s-base -1 --h1 ${hic_dir}/L1_R1.fastq.gz --h2 ${hic_dir}/L1_R2.fastq.gz ${pb_dir}/*fasta.gz


awk '/^S/{print ">"$2"\n"$3}' vpTaxBacc_B34-6.hifiasm.asm.hic.p_ctg.gfa | fold > vpTaxBacc_B34-6.hifiasm.asm.hic.p_ctg_l${l}.fasta
awk '/^S/{print ">"$2"\n"$3}' vpTaxBacc_B34-6.hifiasm.asm.hic.hap1.p_ctg.gfa | fold > vpTaxBacc_B34-6.hifiasm.asm.hic.hap1_l${l}.fasta
awk '/^S/{print ">"$2"\n"$3}' vpTaxBacc_B34-6.hifiasm.asm.hic.hap2.p_ctg.gfa | fold > vpTaxBacc_B34-6.hifiasm.asm.hic.hap2_l${l}.fasta

gfastats vpTaxBacc_B34-6.hifiasm.asm.hic.p_ctg_l${l}.fasta > vpTaxBacc_B34-6.hifiasm.asm.hic.p_ctg_l${l}.stats.txt
gfastats vpTaxBacc_B34-6.hifiasm.asm.hic.hap1_l${l}.fasta > vpTaxBacc_B34-6.hifiasm.asm.hic.hap1_l${l}.stats.txt
gfastats vpTaxBacc_B34-6.hifiasm.asm.hic.hap2_l${l}.fasta > vpTaxBacc_B34-6.hifiasm.asm.hic.hap2_l${l}.stats.txt

cp vpTaxBacc_B34-6.hifiasm.asm.hic.p_ctg_l${l}.fasta ${out_dir}/
cp vpTaxBacc_B34-6.hifiasm.asm.hic.hap1_l${l}.fasta ${out_dir}/
cp vpTaxBacc_B34-6.hifiasm.asm.hic.hap2_l${l}.fasta ${out_dir}/

cp vpTaxBacc_B34-6.hifiasm.asm.hic.p_ctg_l${l}.stats.txt ${out_dir}/
cp vpTaxBacc_B34-6.hifiasm.asm.hic.hap1_l${l}.stats.txt ${out_dir}/
cp vpTaxBacc_B34-6.hifiasm.asm.hic.hap2_l${l}.stats.txt ${out_dir}/
```

# 2. purge_dups on both haplotypes

```bash
purgedups_path=/projects/dazzler/pippel/prog/purge_dups
export PATH=${purgedups_path}/scripts/:$PATH

cores=36
partition=main
time=48:00:00
mem=250G

# create a list of PacBio files
pb_dir=../../data/raw-data/PacBio
ls ${pb_dir}/pb.fasta.gz > pb.fofn

set -e
set -u
set -o pipefail

for x in vpTaxBacc_B34-6.hifiasm.asm.hic.hap[12]_l2.fasta
do
    if [[ -f ${x} ]]
    then
        pd_config.py -B ${x} pb.fofn -n ${x%.*}.config.json
        ## modify config to speed up mapping step
        sed -e "s/\"mnmp_opt\": \"\",/\"mnmp_opt\": \" -x map-hifi\",/" -i ${x%.*}.config.json
        sleep 1s
        cmd="run_purge_dups.py -p bash ${x%.*}.config.json ${purgedups_path}/src ${x%.*}"
        sbatch -c ${cores} -n 1 -p ${partition} -J purgedups --time=${time} --mem=${mem} --wrap="${cmd}"
    fi
done
```

# 5. manual curation

The manual curation was done with PretextView (version 1.0.3) and GRIT_Rapid (git: 1a3d79a8).
The first manual curation round was done in dual-scaffolding mode, i.e. both haplotype assemblies were combined and
a HiC map (pretext) was created. This round was used to roughly organize scaffolds into chromosomes, correct potential
haplotype miss-placements and identifying of corresponding haplotypes.
Afterwards the haplotypes were thoroughly curared individually by incorporating all information from tiara, mitohifi and blobtools.

```bash
# 1. create HiC map with pretext curation pipeline
WORKFLOW=sanger-tol/curationpretext
PROFILE=mpi
WORKDIR=work
RESULTS=results

# cat workflow_parameters.yml
#cram: './hic_cram/'
#input: './comb.fa'
#reads: '../../data/raw-data/PacBio'
#teloseq: 'AAACCCT'
#aligner: 'minimap2'
#outdir: 'results'

nextflow run "$WORKFLOW" \
    $REMOTE_OPTS \
    -profile "$PROFILE" \
    -work-dir "$WORKDIR" \
    -resume \
    -ansi-log false \
    -params-file workflow_parameters.yml \
    --outdir "$RESULTS"

# 2. curate assembly with PretextView
conda activate rapidCuration
pretext-to-tpf -l DEBUG -a comb.fa -p pretext_rerun_normal.pretext.agp_1 -o vpTaxBacc_B34-6_comb_l2_cur01.tpf

~/prog/sanger/rapid-curation/rapid_join.pl -fa comb.fa -tpf vpTaxBacc_B34-6_comb_l2_cur01.hap1.tpf -out vpTaxBacc_B34-6_comb_l2_cur01_hap1
~/prog/sanger/rapid-curation/rapid_join.pl -fa comb.fa -tpf vpTaxBacc_B34-6_comb_l2_cur01.hap2.tpf -out vpTaxBacc_B34-6_comb_l2_cur01_hap2
~/prog/sanger/rapid-curation/rapid_join.pl -fa comb.fa -tpf vpTaxBacc_B34-6_comb_l2_cur01.haplotigs.tpf -out vpTaxBacc_B34-6_comb_l2_cur01_haplotigs
~/prog/sanger/rapid-curation/rapid_join.pl -fa comb.fa -tpf vpTaxBacc_B34-6_comb_l2_cur01.tpf -out vpTaxBacc_B34-6_comb_l2_cur01

# 3. recreate phased assemblies
### rename hap1/hap2 fasta ids
sed -e "s:SUPER_:h1su:" vpTaxBacc_B34-6_comb_l2_cur01.hap1.inter.fa > vpTaxBacc_B34-6_comb_l2_cur01.hap1.renamed.fa
sed -e "s:SUPER_:h2su:" vpTaxBacc_B34-6_comb_l2_cur01.hap2.inter.fa > vpTaxBacc_B34-6_comb_l2_cur01.hap2.renamed.fa

### remove dups/mito/bact etc contigs from shrapnel contigs, that were not treated so far
seqkit split -i vpTaxBacc_B34-6_comb_l2_cur01.inter.fa
### remove mito hits - detected via mitohifi
for x in $(cat hap1_mito_cids.txt hap2_mito_cids.txt)
do
 if [[ -f vpTaxBacc_B34-6_comb_l2_cur01.inter.fa.split/vpTaxBacc_B34-6_comb_l2_cur01.inter.part_${x}.fa ]]
 then
  echo "discard mito hit $x"
  mv vpTaxBacc_B34-6_comb_l2_cur01.inter.fa.split/vpTaxBacc_B34-6_comb_l2_cur01.inter.part_${x}.fa vpTaxBacc_B34-6_comb_l2_cur01.inter.fa.split/vpTaxBacc_B34-6_comb_l2_cur01.inter.part_${x}.fa.disc
 else
  echo "could not find mito hit $x"
 fi
done

### remove tiara hits
for t in bacteria prokarya organelle
do
 for x in $(grep -h -e ${t} hap1_tiara_cids.txt hap2_tiara_cids.txt | awk '{print $1}')
 do
  if [[ -f vpTaxBacc_B34-6_comb_l2_cur01.inter.fa.split/vpTaxBacc_B34-6_comb_l2_cur01.inter.part_${x}.fa ]]
  then
   echo "discard ${t} hit $x (TIARA)"
   mv vpTaxBacc_B34-6_comb_l2_cur01.inter.fa.split/vpTaxBacc_B34-6_comb_l2_cur01.inter.part_${x}.fa vpTaxBacc_B34-6_comb_l2_cur01.inter.fa.split/vpTaxBacc_B34-6_comb_l2_cur01.inter.part_${x}.fa.disc
  else
   echo "could not find ${t} hit $x (TIARA)"
  fi
 done
done

## create haplotype assemblies
( cat vpTaxBacc_B34-6_comb_l2_cur01.hap1.renamed.fa &&
cat vpTaxBacc_B34-6_comb_l2_cur01.inter.fa.split/vpTaxBacc_B34-6_comb_l2_cur01.inter.part_scaff*fa vpTaxBacc_B34-6_comb_l2_cur01.inter.fa.split/vpTaxBacc_B34-6_comb_l2_cur01.inter.part_h1tg*fa | seqkit sort -r -l
) > vpTaxBacc_B34-6_comb_l2_cur01.hap1.filt.fa

( cat vpTaxBacc_B34-6_comb_l2_cur01.hap2.renamed.fa &&
cat vpTaxBacc_B34-6_comb_l2_cur01.inter.fa.split/vpTaxBacc_B34-6_comb_l2_cur01.inter.part_scaff*fa vpTaxBacc_B34-6_comb_l2_cur01.inter.fa.split/vpTaxBacc_B34-6_comb_l2_cur01.inter.part_h2tg*fa | seqkit sort -r -l
) > vpTaxBacc_B34-6_comb_l2_cur01.hap2.filt.fa

## continue with the same steps 1.-3. separately for vpTaxBacc_B34-6_comb_l2_cur01.hap1.filt.fa  and vpTaxBacc_B34-6_comb_l2_cur01.hap1.filt.fa
```
